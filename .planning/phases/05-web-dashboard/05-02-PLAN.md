```rust
pub async fn search_handler(
    State(state): State<Arc<AppState>>,
    Query(params): Query<SearchParams>,
) -> Result<SearchTemplate, AppError> {
    let mode_str = params.mode.unwrap_or_else(|| "hybrid".to_string());
    let query = params.q.clone().unwrap_or_default();

    // If no query, return empty form
    if query.is_empty() {
        return Ok(SearchTemplate {
            query: None,
            mode: mode_str,
            results: vec![],
            result_count: 0,
            active_nav: "search",
        });
    }

    // Parse mode string to SearchMode enum
    let search_mode = match mode_str.as_str() {
        "semantic" => crate::search::SearchMode::Semantic,
        "fts" => crate::search::SearchMode::Fts,
        _ => crate::search::SearchMode::Hybrid,
    };

    // Construct SearchEngine per-request (cheap -- just two references)
    let engine = crate::search::SearchEngine::new(&state.store, &*state.embedder);

    let opts = crate::search::SearchOptions {
        query: query.clone(),
        limit: 20,
        min_score: None,
        mode: search_mode,
        path_filter: None,
        tag_filter: None,
        context: 0,
    };

    let response = engine.search(&opts).await?;

    let results: Vec<SearchResultView> = response
        .results
        .into_iter()
        .filter(|r| r.is_context != Some(true))  // Skip context chunks for display
        .map(|r| {
            // Truncate chunk text to ~300 chars for preview
            let preview = if r.chunk_text.len() > 300 {
                let mut end = 300;
                // Don't cut in the middle of a multi-byte character
                while !r.chunk_text.is_char_boundary(end) && end < r.chunk_text.len() {
                    end += 1;
                }
                format!("{}...", &r.chunk_text[..end])
            } else {
                r.chunk_text.clone()
            };
            SearchResultView {
                file_path: r.file_path,
                heading_breadcrumb: r.heading_breadcrumb,
                chunk_text: preview,
                similarity_score: (r.similarity_score * 100.0).round() / 100.0,
            }
        })
        .collect();

    let result_count = results.len();

    Ok(SearchTemplate {
        query: Some(query),
        mode: mode_str,
        results,
        result_count,
        active_nav: "search",
    })
}
```

{% extends "base.html" %}
{% block title %} -- Search{% endblock %}

    name="q"

    value="{{ query.as_deref().unwrap_or_default() }}"
    class="search-input"
    aria-label="Search query"
  >
  <select name="mode" class="search-mode" aria-label="Search mode">
    <option value="hybrid"{% if mode == "hybrid" %} selected{% endif %}>hybrid</option>
    <option value="semantic"{% if mode == "semantic" %} selected{% endif %}>semantic</option>
    <option value="fts"{% if mode == "fts" %} selected{% endif %}>fts</option>
  </select>
  <button type="submit" class="search-button">Search Notes</button>
</form>

{% if let Some(ref q) = query %}
  {% if result_count > 0 %}
    <p class="results-summary">Showing {{ result_count }} results for "{{ q }}" ({{ mode }} mode)</p>
    {% for result in results %}
    <div class="result-card">
      <div class="result-header">
        <span class="result-path">{{ result.file_path }}{% if !result.heading_breadcrumb.is_empty() %} &rsaquo; {{ result.heading_breadcrumb }}{% endif %}</span>
        <span class="result-score">{{ result.similarity_score }}</span>
      </div>
      <div class="result-body">{{ result.chunk_text }}</div>

      <h2>No matching notes found</h2>

  {% endif %}
{% endif %}
{% endblock %}
