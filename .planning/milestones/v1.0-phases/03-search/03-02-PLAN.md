```rust
cli::Command::Search {
    query, limit, min_score, mode, path_filter, tag_filter, context, format,
} => {
    // Resolve data directory (same logic as index command)
    let data_dir = cli.data_dir.clone()
        .unwrap_or_else(|| std::env::current_dir().unwrap().join(".local-index"));
    let db_path = data_dir.to_string_lossy().to_string();

    // Open store
    let store = ChunkStore::open(&db_path).await?;

    // For semantic/hybrid modes, we need an embedder
    let needs_embedder = matches!(mode, cli::SearchMode::Semantic | cli::SearchMode::Hybrid);

    // Resolve credentials and create embedder only if needed
    let api_key;
    let embedder;
    let embedder_ref: &dyn Embedder;

    if needs_embedder {
        api_key = resolve_voyage_key()?;
        embedder = VoyageEmbedder::new(api_key);
        embedder_ref = &embedder;
    } else {
        // FTS-only mode: create a dummy embedder that will never be called
        // Better: make embedder optional in SearchEngine, or just require credentials always
        // Simplest for v1: always require credentials (user must have VOYAGE_API_KEY set)
        api_key = resolve_voyage_key()?;
        embedder = VoyageEmbedder::new(api_key);
        embedder_ref = &embedder;
    }

    let engine = SearchEngine::new(&store, embedder_ref);

    // Convert CLI SearchMode to library SearchMode
    let lib_mode = match mode {
        cli::SearchMode::Semantic => LibSearchMode::Semantic,
        cli::SearchMode::Fts => LibSearchMode::Fts,
        cli::SearchMode::Hybrid => LibSearchMode::Hybrid,
    };

    let opts = SearchOptions {
        query: query.clone(),
        limit: *limit,
        min_score: *min_score,
        mode: lib_mode,
        path_filter: path_filter.clone(),
        tag_filter: tag_filter.clone(),
        context: *context,
    };

    let response = engine.search(&opts).await?;

    // Format and output
    let output = match format {
        cli::OutputFormat::Json => format_json(&response)
            .map_err(|e| anyhow::anyhow!("Failed to serialize results: {}", e))?,
        cli::OutputFormat::Pretty => format_pretty(&response),
    };

    println!("{}", output);
}
```

**3. Add FTS index creation to the index command.** After the successful index loop completes (after the "indexing complete" tracing::info! call), add:

```rust
// Create/refresh FTS index for search command
if chunks_embedded > 0 {
    tracing::info!("creating FTS index for search");
    let engine = local_index::search::SearchEngine::new(&store, &embedder);
    if let Err(e) = engine.ensure_fts_index().await {

        tracing::info!("FTS index created successfully");

This ensures the FTS index is ready after indexing. If it fails, search will create it lazily on first use (ensure_fts_index is called in fts_search and hybrid_search).

use local_index::pipeline::embedder::Embedder;
use local_index::types::EmbeddingResult;
use local_index::error::LocalIndexError;

struct MockEmbedder;

impl Embedder for MockEmbedder {
    async fn embed(&self, texts: &[String]) -> Result<EmbeddingResult, LocalIndexError> {
        // Return deterministic embeddings based on text content.
        // Use a simple hash-to-vector approach: each text gets a slightly different vector.
        let embeddings: Vec<Vec<f32>> = texts.iter().enumerate().map(|(i, _text)| {
            let mut vec = vec![0.0f32; 1024];
            // Put a 1.0 at a position based on index to make vectors distinguishable
            vec[i % 1024] = 1.0;
            // Add some base similarity so cosine distance is reasonable
            for v in vec.iter_mut() {
                *v += 0.1;
            }
            vec
        }).collect();
        Ok(EmbeddingResult {
            embeddings,
            model: "mock-model".to_string(),
            total_tokens: texts.len() as u64 * 10,
        })
    }

    fn model_id(&self) -> &str { "mock-model" }
    fn dimensions(&self) -> usize { 1024 }
}
```
